class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
        vector<int>ans;
        map<int, int> mpp;
        int n = nums.size();
        for(int i = 0; i < n; i++)
        {
            mpp[nums[i]]++;
            if(ans.size() == 0 || ans[0] != nums[i])
            {
                if(mpp[nums[i]] > n/3)
                    ans.push_back(nums[i]);
            }
                
            if(ans.size() == 2) break;
        }
        return ans;
    }
};
