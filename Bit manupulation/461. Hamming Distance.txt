class Solution {
public:
    int hammingDistance(int x, int y) {
        int count = 0;
        for(int i = 0; i < 32; i++)
        {
            if(((x & 1 << i) && !(y & 1<<i)) || (!(x&1<<i) && (y & 1 << i)))
                count++;
        }
        return count;
    }
};

class Solution {
public:
    int hammingDistance(int x, int y) {
        int xored = x^y;
        return bitset<32>(xored).count();
    }
};
